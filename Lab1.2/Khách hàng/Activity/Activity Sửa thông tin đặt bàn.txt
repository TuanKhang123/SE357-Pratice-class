@startuml
start

  repeat
    :Đăng nhập;
  repeat while (Đăng nhập thành công?) is (no)
    ->yes;

  if (Có bàn?) then (yes)
    :Chọn bàn cần chỉnh sửa thông tin;
      :Điền thông tin chỉnh sửa;
      repeat
        :Ấn "Cập nhật";
      repeat while (Kiểm tra thông tin?) is (không hợp lệ)
        ->hợp lệ;
        :Lưu thông tin chỉnh sửa bàn;
        :Hiển thị thông báo chỉnh sửa thành công;
  else (no)
    :Hiển thị thông báo không có bàn để chỉnh sửa;
  endif


stop
@enduml
